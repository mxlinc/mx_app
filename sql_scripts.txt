CREATE TABLE prod.user_table (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    enabled BOOLEAN NOT NULL DEFAULT TRUE,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(50) NOT NULL,         -- e.g. 'student' or 'admin'
    name VARCHAR(150),                 -- Full name of the user
    level_code VARCHAR(50),            -- e.g. grade level or custom code
    hint VARCHAR(255)                  -- password hint or other notes
);
